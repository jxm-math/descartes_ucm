\begin{frame}\frametitle{1.1 La Geometría de Descartes}
	
Consideremos la ecuación algebraica de grado $n$:
	
$$P(x)\equiv a_n x^n + a_{n-1}x^{n-1} + \cdots + a_0 =0$$
	
donde los coeficientes $a_k$ son reales y $a_n\neq 0$\\~\\
	
\pause
	
\textbf{Teorema Fundamental del Álgebra} Toda ecuación algebraica de grado $n$ tiene $n$ raíces reales o complejas, contadas según su multiplicidad
	
\end{frame}

\begin{frame}
	
Una raíz $\xi$ de la ecuación $P(x)=0$ tiene multiplicidad $s$ cuando 
	
$$P(\xi)=P'(\xi)=P''(\xi)=\cdots=P^{s-1)}(\xi)=0$$ 
$$P^{s)}(\xi)\neq 0$$ 

\end{frame}

\begin{frame}
	
\textbf{Teorema} Si los coeficientes de la ecuación algebraica son reales, sus raíces son reales o conjugadas a pares, es decir, si $\xi=\alpha+\ii\beta$, $\alpha,\beta\in\rr$ es raíz de $P(x)$, también lo es $\bar\xi=\alpha-\ii\beta$\\~\\
	
	\pause
	
	\textbf{Corolario} Una ecuación algebraica de grado impar con coeficientes reales tiene al menos una raíz real
	
\end{frame}

\begin{frame}
	
	\textbf{Teorema de Newton} Si para $x=c>0$ el polinomio $P(x)$ y todas sus derivadas $P'(x)$, $P''(x)$, ... , $P^{n)}(x)$ son no negativas
	
	$$P^{k)}(c)\geqslant 0\quad (k=0,1,2,\cdots,n)$$
	
	y $P^{n)}(c)=n!a_n>0$, entonces $R=c$ puede tomarse como frontera superior de las raíces positivas de la ecuación $P(x)=0$\\~\\
	
	\pause
	
	\textbf{Demostración} Si $x>c$, por la fórmula de Taylor,
	
	$P(x)=P(c)+P'(c)(x-c)+\cdots+\frac{P^{n)}}{n!}(x-c)^n>0$
	
	y así todas las raíces positivas $x^+$ satisfacen $x^+\leqslant c$ 
			
\end{frame}

\begin{frame}
	
	En las aplicaciones prácticas del Teorema de Newton se utiliza el sistema de tanteo (por ejemplo, mediante el esquema de Hörner) para hallar una secuencia monótona creciente de números positivos
	
	$$0<c_1\leqslant c_2\leqslant\cdots\leqslant c_{n-1}\leqslant c_n$$
	
	para los cuales se cumplan las desigualdades
	
	$$P^{n-1)}(c_{1})\geqslant 0$$
	$$P^{n-2)}(c_{2})\geqslant 0$$
	$$\cdots$$
	$$P'(c_{n-1})\geqslant 0$$
	$$P(c_{n})\geqslant 0$$
	
	Tales números existen siempre, pues si $a_n>0$,
	
	$$P^{k)}(x)\xrightarrow{x\longrightarrow\infty}\infty\quad (k=0,1,2,\cdots,n-1)$$
	
	y podemos tomar $c=c_n$
	
\end{frame}

\begin{frame}
	
	En efecto, como $P^{n)}(x)=n!a_n>0$, la función $P^{n-1)}(x)$ es creciente, y así
	
	$$P^{n-1)}(x)>P^{n-1)}(c_1)\geqslant 0$$
	
	\pause
	
	Así de nuevo, $P^{n-2)}(x)$ es creciente en el intervalo $[c_1,\infty)$, y por tanto, para $x>c_2\geqslant c_1$,
	
	$$P^{n-2)}(x)>P^{n-2)}(c_2)\geqslant 0$$
	
	\pause
	
	Iterando este razonamiento, llegamos a que $P(x)$ es creciente en el intervalo $[c_{n-1},\infty)$, y para $x>c_n\geqslant c_{n_1}$
	
	$$P(x)>P(c_n)\geqslant 0$$
	
	con lo que $x^+\leqslant c_n$
	
\end{frame}

\begin{frame}
	
	\textbf{Ejemplo} Sea
	
	$$P(x)=2x^5-100x^2+2x-1$$
	
	\pause
	
	En este caso,
	
	$$P'(x)=10x^4-200x+2$$
	$$P''(x)=40x^3-200$$
	$$P'''(x)=120x^2$$
	$$P^{4)}(x)=240x$$
	$$P^{5)}=240$$
	
\end{frame}

\begin{frame}
	
	$P'''(x)>0$, $P^{4)}(x)>0$, $P^{5)}(x)>0$ para $x>0$\\~\\
	
	También
	
	$$P''(x)=40(x^3-5)>0\quad\text{para }x\geqslant 2$$
	
	Supondremos $c_1=c_2=c_3=2$\\~\\
	
	\pause
	
	$P'(2)=-238<0$, pero $P'(3)=212>0$ así que tomamos $c_4=3$\\~\\
	
	\pause
	
	$P(3)=-409<0$, pero $P(4)=455>0$ así que tomamos $c_5=4$, y $R=4$ es frontera superior de las raíces positivas de esta ecuación 
	
	\hfill{\small \href{http://mathifold.org/es/visualisation_classic_spirals.html}{[Espirales Clásicas]}}
	
\end{frame}

\begin{frame}
	
	¿Cómo hallar el número total de raíces positivas de una ecuación algebraica? Recordemos
	
	\begin{itemize}
		\item Si $P(a)P(b)<0$, en el intervalo $(a,b)$ existe un número impar de raíces de $P(x)$ (contando multiplicidades)
		\item Si $P(a)P(b)>0$, en el intervalo $(a,b)$ existe un número par (o nulo) de raíces de $P(x)$
	\end{itemize}
	
\end{frame}

\begin{frame}
	
	\textbf{Teorema de Sturm} Si un polinomio $P(x)$ no tiene raíces múltiples y $P(a)\,P(b)\neq 0$, el número de sus raíces reales en el intervalo $a<x<b$ es exactamente igual al número de cambios de signo perdidos en la secuencia de Sturm del polinomio $P(x)$ yendo de $x=a$ a $x=b$, es decir
	
	$$N(a,b)=N(a)-N(b)$$ 
	
	\pause
	
	\textbf{Corolario 1} Si $P(0)\neq 0$, los números $N_+$ y $N_-$ de raíces positivas y negativas del polinomio $P(x)$ son
	
	$$N_+=N(0)-N(+\infty)\qquad N_-=N(-\infty)-N(0)$$
	
\end{frame}

\begin{frame}
	
	\textbf{Corolario 2} Un polinomio $P(x)$ de grado $n$ (sin raíces múltiples) tiene todas sus raíces reales si y sólo si
	
	$$N(-\infty)-N(+\infty)=n$$
	
	De este modo, si $a_n>0$, todas la raíces serán reales si y sólo si
	
	\begin{enumerate}
		\item la secuencia de Sturm tiene un número máximo de elementos $n+1$, esto es, $m=n$
		\item son ciertas las desigualdades $P_k(+\infty)>0$, es decir, los coeficientes dominantes de todas las funciones de Sturm $P_k(x)$ son positivos 
	\end{enumerate}
	
\end{frame}

\begin{frame}
	
	\textbf{Ejemplo} Determinar el número de raíces positivas y negativas de la ecuación
	
	$$x^4-4x+1=0$$
	
	\pause
	
	\textbf{Solución} La secuencia de Sturm es de la forma
	
	$$P(x)=x^4-4x+1$$
	$$P_1(x)=x^3-1$$
	$$P_2(x)=3x-1$$
	$$P_3(x)=1$$

	de donde
	
	$$N(-\infty)=2\quad N(0)=2\quad N(+\infty)=0$$
	$$N_+=2-0=2\qquad N_-=2-2=0$$
	
	con lo que $P(x)$ tiene dos raíces positivas y ninguna negativa (y por tanto dos raíces complejas)
	
\end{frame}

\begin{frame}
	
	\textbf{Teorema de Budan-Fourier} Si los números $a<b$ no son raíces de un polinomio $P(x)$ de grado $n$, el número $N(a,b)$ de raíces reales de la ecuación $P(x)=0$ que caen entre $a$ y $b$ es igual al número mínimo $\Delta N$ de pérdidas de cambio de signo en la secuencia de derivadas sucesivas
	
	$$P(x),\,P'(x),\,\cdots,\,P^{n-1)}(x),\,P^{n)}(x)$$
	
	yendo de $x=a$ a $x=b$, o menor que $\Delta N$ por un número par
	
	$$N(a,b)=\Delta N-2k$$
	
	donde 
	
	$$\Delta N=\ubar{N}(a)-\bar{N}(b)$$
	
	y $\ubar{N}(a)$, $\bar{N}(b)$ son los números inferior y superior de variaciones de signo en la secuencia anterior en $x=a,\,b$
	
\end{frame}

\begin{frame}
	
	\textbf{Teorema de Hua} Si una ecuación 
	
	$$a_n x^n + a_{n-1}x^{n-1} + \cdots + a_0 =0$$
	
	tiene coeficientes reales y todas sus raíces son reales, el cuadrado de cada coeficiente no extremo de la ecuación es mayor que el producto de los dos coeficientes adyacentes
	
	$$a_k^2>a_{k-1}a_{k+1}$$
	
	\pause
	
	\textbf{Corolario} Si para cierto $k$ tenemos la desigualdad 
	
	$$a_k^2\leqslant a_{k-1}a_{k+1}$$
	
	entonces la ecuación tiene al menos un par de raíces complejas
	
\end{frame}

\begin{frame}
	
	\textbf{Ejemplo} Determínese la composición de las raíces de la ecuación 
	
	$$x^4+8x^3-12x^2+104x-20=0$$
	
	\pause
	
	\textbf{Solución} Como
	
	$$(-12)^2<8\cdot 104$$
	
	la ecuación tiene raíces complejas, y el número de raíces reales no excede de 2. En la secuencia de coeficientes de la ecuación hay $\Delta N=3$ variaciones de signo y $\Delta P=1$ no variaciones de signo. Así, en base a la regla de Descartes y a sus corolarios, nuestra ecuación tiene una raíz positiva, una negativa y un par de raíces complejas
	
\end{frame}

\begin{frame}
	
	\textbf{Método de Bernouilli}  Supongamos que se tiene una ecuación algebraica
	
	$$a_n x^n + a_{n-1}x^{n-1} + \cdots + a_0 =0$$
	
	cuyas raíces $x_1,\,x_2,\,\cdots x_n$ son distintas\\~\\
	
	\pause
	
	En base a los coeficientes $a_k$ formamos la relación de recurrencia
	
	$$a_n y_{n+i} + a_{n-1}y_{n-1+i} + \cdots + a_0y_i =0\qquad (i=0,\,1,\,2,\,\cdots)$$
	
\end{frame}

\begin{frame}\frametitle{1.2 Problemas clásicos de Apolonio y Pappus}
	
	La teoría general de segundo grado (enfoque clásico)
	
	$$F(x,y)=a_{11}x^2+2a_{12}xy+a_{22}y^2+2a_{13}x+2a_{23}y+a_{33}=0$$
	
	Comprende las cónicas en un sistema de referencia cartesiana
	
\end{frame}

\begin{frame}
	
	Invariantes por semejanza
	
	$$\Delta=\left|
	\begin{matrix}
	a_{11} & a_{12} &  a_{13}\\
	a_{12} & a_{22} &  a_{23}\\
	a_{13} & a_{23} &  a_{33}\\
	\end{matrix}
	\right|$$
	
	$$\delta=\left|
	\begin{matrix}
	a_{11} & a_{12}\\
	a_{12} & a_{22}\\
	\end{matrix}
	\right|$$
	
	$\Delta\neq 0 \quad\leadsto$ curva no degenerada
	
	$\Delta= 0 \quad\leadsto$ curva degenerada
	
\end{frame}

\begin{frame}
	
	\textbf{CASO I:} $\delta\neq 0$
	
	Se toma la transformación de coordenadas
	
	$$
	\left\{
	\begin{array}{ccc}
	x & = & x'+u\\
	y & = & y'+v\\
	\end{array}
	\right.\Longleftrightarrow
	\left\{
	\begin{array}{ccc}
	x' & = & x-u\\
	y' & = & y-v\\
	\end{array}
	\right.
	$$
	
	$$\dfrac{\partial F}{\partial x}=2a_{11}x+2a_{12}y+2a_{13}=0$$
	
	$$\dfrac{\partial F}{\partial y}=2a_{12}x+2a_{22}y+2a_{23}=0$$
	
	Tomamos $x'=y'=0$
	
	$$
	\left\{
	\begin{array}{cccccc}
	a_{11}u & + &a_{12}v & + & a_{13} & = 0\\
	a_{12}u & + &a_{22}v & + & a_{23} & = 0\\
	\end{array}
	\right.
	$$
	
	$$
	\begin{pmatrix}
	a_{11} & a_{12}\\
	a_{12} & a_{22}\\
	\end{pmatrix}
	\begin{pmatrix}
	u\\
	v\\
	\end{pmatrix}
	=-
	\begin{pmatrix}
	a_{13}\\
	a_{23}\\
	\end{pmatrix}
	$$
	
\end{frame}

\begin{frame}
	
	$$
	\begin{pmatrix}
	u\\
	v\\
	\end{pmatrix}
	=-\dfrac{1}{\delta}
	\begin{pmatrix}
	a_{22} & -a_{12}\\
	-a_{12} & a_{11}\\
	\end{pmatrix}
	\begin{pmatrix}
	a_{13}\\
	a_{23}\\
	\end{pmatrix}
	$$
	
	Al insertar en la ecuación inicial nos queda
	
	$$F(x',y')=a_{11}x'^2+2a_{12}x'y'+a_{22}y'^2+\dfrac{\Delta}{\delta}=0$$
	
	\textbf{obs:} $(u,v)$ corresponde al centro de la cónica
	
\end{frame}

\begin{frame}
	
	Rotación de la referencia $(x',y')$
	
	$$
	\begin{pmatrix}
	x'\\
	y'\\
	\end{pmatrix}
	=
	\begin{pmatrix}
	\cos\alpha & -\sin\alpha\\
	\sin\alpha & \cos\alpha\\
	\end{pmatrix}
	\begin{pmatrix}
	\xi\\
	\eta\\
	\end{pmatrix}
	$$
	
	En la referencia $(\xi,\eta)$, nuestra ecuación toma la forma
	
	$$F(\xi,\eta)=b_{11}\xi^2+b_{22}\eta^2+\dfrac{\Delta}{\delta}=0$$
	
	donde
	
	$$b_{11}=\dfrac{1}{2}\left(a_{11}+a_{22}\pm\sqrt{(a_{11}-a_{22})^2+4a_{12}^2}\right)$$
	
	$$b_{11}=\dfrac{1}{2}\left(a_{11}+a_{22}\mp\sqrt{(a_{11}-a_{22})^2+4a_{12}^2}\right)$$
	
	\textbf{obs:} 
	
	$a_{12}>0\,\leadsto$ los signos de la raíz son $(b_{11}\rightarrow +,b_{22}\rightarrow -)$
	
	$a_{12}<0\,\leadsto$ los signos de la raíz son $(b_{11}\rightarrow -,b_{22}\rightarrow +)$ 
	
	$\delta=b_{11}b_{22}$, $b_{11}+b_{22}=a_{11}+a_{22}$
	
\end{frame}

\begin{frame}
	
	\textbf{a)} $\delta>0$, $\Delta\neq 0$
	
	$$\text{elipse real }\left\{
	\begin{array}{cc}
	b_{11} & >0\\
	b_{22} & >0\\
	\Delta & <0\\
	\end{array}
	\right.\qquad
	\text{elipse imaginaria }\left\{
	\begin{array}{cc}
	b_{11} & >0\\
	b_{22} & >0\\
	\Delta & >0\\
	\end{array}
	\right.$$
	
	\textbf{b)} $\delta>0$, $\Delta= 0$ rectas coincidentes imaginarias
	
\end{frame}

\begin{frame}
	
	\textbf{c)} $\delta<0$, $\Delta\neq 0$
	
	$$\text{hipérbola }\left\{
	\begin{array}{cc}
	b_{11} & >0\\
	b_{22} & <0\\
	\Delta & \neq 0\\
	\end{array}
	\right.$$
	
	\textbf{d)} $\delta<0$, $\Delta= 0$ rectas coincidentes reales
	
\end{frame}

\begin{frame}
	
	\textbf{CASO II:} $\delta= 0$
	
	Rotación de la referencia $(x,y)$
	
	$$
	\begin{pmatrix}
	x\\
	y\\
	\end{pmatrix}
	=
	\begin{pmatrix}
	\cos\alpha & -\sin\alpha\\
	\sin\alpha & \cos\alpha\\
	\end{pmatrix}
	\begin{pmatrix}
	x'\\
	y'\\
	\end{pmatrix}
	$$
	
	En la referencia $(x',y')$ la ecuación toma la forma
	
	$$F(x',y')=b_{11}x'^2+2b_{13}x'+2b_{23}y'+a_{33}=0$$
	
	con
	
	$b_{11}=\mathrm{tr}$, $b_{13}=\dfrac{a_{13}\sqrt{a_{11}}+a_{23}\sqrt{a_{22}}}{\sqrt{\mathrm{tr}}}$, $b_{23}=\dfrac{a_{23}\sqrt{a_{11}}+a_{13}\sqrt{a_{22}}}{\sqrt{\mathrm{tr}}}$
					
\end{frame}
	
\begin{frame}
		
	Tomamos el cambio de referencia
	
	$$x'=\xi-\dfrac{b_{13}}{b_{11}}\qquad y'=\eta-\dfrac{b_{11}a_{33}-b_{13}^2}{b_{11}}$$ 
	
	y llegamos a 
	
	$$F(\xi,\eta)=\xi^2+2\dfrac{b_{23}}{b_{11}}\eta=0$$
					
\end{frame}

\begin{frame}
	
	\textbf{a)} $\Delta\neq 0$ parábola
	
	\textbf{b)} $\Delta= 0$ 
	
	$$
	\begin{array}{rcl}
		\text{rectas reales paralelas} & \text{si} & a_{13}^2-a_{11}a_{33}>0\\
		\text{recta doble} & \text{si} & a_{13}^2-a_{11}a_{33}=0\\
		\text{rectas imaginarias paralelas} & \text{si} & a_{13}^2-a_{11}a_{33}<0\\
	\end{array}
	$$
	
\end{frame}

	\begin{center}
		\includegraphics[scale=0.5]{conic_classification.pdf}
	\end{center}
	
\end{frame}

\begin{frame}
	
	\textbf{Ejemplo} $F(x,y)=x^2+2xy+y^2+1=0$
	
	$$\Delta=\left|
	\begin{matrix}
	1 & 1 &  0\\
	1 & 1 &  0\\
	0 & 0 &  1\\
	\end{matrix}
	\right|=0$$
	
	$$\delta=\left|
	\begin{matrix}
	1 & 1\\
	1 & 1\\
	\end{matrix}
	\right|=0$$
	
	$$a_{13}^2-a_{11}a_{33}=-1<0$$
	
	par de rectas paralelas imaginarias
	
\end{frame}

\begin{frame}
	
	\textbf{Ejemplo} $F(x,y)=5x^2+4xy+2y^2-18x-12y+15=0$\\~\\
	
	elipse, centro $(1,2)$ en la referencia original
	
	$$F(x',y')=5x'^2+4x'y'+2y'^2-6=0$$
	
	ángulo de giro: $\tan(2\alpha)=\dfrac{4}{3}$
	
\end{frame}
